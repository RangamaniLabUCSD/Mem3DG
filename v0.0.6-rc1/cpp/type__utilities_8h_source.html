96 template <typename DerivedA, typename DerivedB>
97 auto rowwiseDotProduct(const Eigen::DenseBase<DerivedA> &A,
98  const Eigen::DenseBase<DerivedB> &B) {
99  EIGEN_STATIC_ASSERT_SAME_MATRIX_SIZE(DerivedA, DerivedB);
100  return (A.derived().array() * B.derived().array()).rowwise().sum().matrix();
101 }

112 template <typename VectorType, typename Derived>
113 auto rowwiseScalarProduct(const Eigen::DenseBase<VectorType> &A,
114  const Eigen::DenseBase<Derived> &B) {
115  EIGEN_STATIC_ASSERT_VECTOR_ONLY(VectorType);
116  if (A.rows() != B.rows()) {
117  mem3dg_runtime_error("Mismatched row size, ", A.rows(), " rows in vector ",
118  B.rows(), " in matrix, for rowwise scalar product");
119  }
120  return (B.derived().array().colwise() * A.derived().array()).matrix();
121 }

135 template <typename Derived>
136 typename Derived::PlainMatrix
137 rowwiseCrossProduct(const Eigen::DenseBase<Derived> &A,
138  const Eigen::DenseBase<Derived> &B) {
139  if (A.rows() != B.rows()) {
140  mem3dg_runtime_error("Mismatched rows, ", A.rows(), " rows in A", B.rows(),
141  " in B, for rowwise cross product");
142  }
143  typename Derived::PlainMatrix C;
144  C.resize(A.rows(), 3);
145  for (std::size_t i = 0; i < A.rows(); i++) {
146  C.row(i) = (A.derived().matrix().row(i)).cross(B.derived().matrix().row(i));
147  }
148  return C;
149 }
