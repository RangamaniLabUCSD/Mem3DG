616  template <typename T, std::size_t k>
617  EigenVectorXkr_T<T, k> getVar(const nc::NcVar &var,
618  const std::size_t idx) const {
619  assert(idx < nFrames());
620 
621  nc_vlen_t vlenData;
622  var.getVar({idx}, &vlenData);
623 
624  // Initialize an Eigen object and copy the data over
625  EigenVectorXkr_T<T, k> vec(vlenData.len / k, k);
626  // Bind to nc_vlen_t memory and copy data over
627  vec = AlignedEigenMap_T<T, k, Eigen::RowMajor>(static_cast<T *>(vlenData.p),
628  vlenData.len / k, k);
629  return vec;
630  }

691  void initializeConventions() {
692 
693  const int compression_level = 5;
694 
695  // initialize data
696  fd->putAtt(CONVENTIONS_NAME, CONVENTIONS_VALUE);
697  fd->putAtt(CONVENTIONS_VERSION_NAME, CONVENTIONS_VERSION_VALUE);
698  fd->putAtt(MEM3DG_VERSION_NAME, MEM3DG_VERSION);
699 
700  parameter_group = fd->addGroup(PARAM_GROUP_NAME);
701  traj_group = fd->addGroup(TRAJ_GROUP_NAME);
702 
703  frame_dim = traj_group.addDim(FRAME_NAME);
704 
705  time_var = traj_group.addVar(TIME_VAR, netCDF::ncDouble, {frame_dim});
706  time_var.putAtt(UNITS, TIME_UNITS);
707  time_var.setCompression(true, true, compression_level);
708 
709  uint_array_t = traj_group.addVlenType(UINT_ARR, nc::ncUint);
710  double_array_t = traj_group.addVlenType(DOUBLE_ARR, nc::ncDouble);
711  bool_array_t = traj_group.addVlenType(BOOL_ARR, nc::ncByte);
712 
713  topo_var = traj_group.addVar(TOPO_VAR, uint_array_t, {frame_dim});
714  coord_var = traj_group.addVar(COORD_VAR, double_array_t, {frame_dim});
715  refcoord_var = traj_group.addVar(REFCOORD_VAR, double_array_t, {frame_dim});
716  vertex_var = traj_group.addVar(VERTEX_VAR, bool_array_t, {frame_dim});
