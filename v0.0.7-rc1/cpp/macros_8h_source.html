33 inline void show_backtrace() {
34  unw_cursor_t cursor;
35  unw_context_t context;
36 
37  // Initialize cursor to current frame for local unwinding.
38  unw_getcontext(&context);
39  unw_init_local(&cursor, &context);
40 
41  // Unwind frames one by one, going up the frame stack.
42  while (unw_step(&cursor) > 0) {
43  unw_word_t offset, ip;
44  // get instruction pointer
45  unw_get_reg(&cursor, UNW_REG_IP, &ip);
46  if (ip == 0) {
47  break;
48  }
49  printf("0x%lx:", ip);
50 
51  char sym[256];
52  if (unw_get_proc_name(&cursor, sym, sizeof(sym), &offset) == 0) {
53  printf(" (%s+0x%lx)\n", sym, offset);
54  } else {
55  printf(" -- error: unable to obtain symbol name for this frame\n");
56  }
57  }
58 }
59 #else
60 #define show_backtrace() ((void)0)
61 #endif
