535  inline gcs::Vertex splitEdge(gcs::Edge e) {
536  gcs::Vertex vertex1 = e.firstVertex(), vertex2 = e.secondVertex();
537  gc::Vector3 vertex1Pos = geometry.vpg->vertexPositions[vertex1];
538  gc::Vector3 vertex2Pos = geometry.vpg->vertexPositions[vertex2];
539  gc::Vector3 vertex1Vel = velocity[vertex1];
540  gc::Vector3 vertex2Vel = velocity[vertex2];
541  double vertex1GeoDist = geometry.geodesicDistance[vertex1];
542  double vertex2GeoDist = geometry.geodesicDistance[vertex2];
543  double vertex1Phi = proteinDensity[vertex1];
544  double vertex2Phi = proteinDensity[vertex2];
545 
546  // bool vertex1PointTracker = geometry.notableVertex[vertex1];
547  // bool vertex2PointTracker = geometry.notableVertex[vertex2];
548 
549  // split the edge
550  gcs::Vertex newVertex = geometry.mesh->splitEdgeTriangular(e).vertex();
551 
552  // update quantities
553  // Note: think about conservation of energy, momentum and angular
554  // momentum
555  // averageData(geometry.vpg->inputVertexPositions, vertex1, vertex2,
556  // newVertex); averageData(velocity, vertex1, vertex2, newVertex);
557  // averageData(geometry.geodesicDistance, vertex1, vertex2, newVertex);
558  // averageData(proteinDensity, vertex1, vertex2, newVertex);
559  geometry.vpg->vertexPositions[newVertex] = 0.5 * (vertex1Pos + vertex2Pos);
560  velocity[newVertex] = 0.5 * (vertex1Vel + vertex2Vel);
561  geometry.geodesicDistance[newVertex] =
562  0.5 * (vertex1GeoDist + vertex2GeoDist);
563  proteinDensity[newVertex] = 0.5 * (vertex1Phi + vertex2Phi);
564  geometry.notableVertex[newVertex] = false;
565  forces.forceMask[newVertex] = gc::Vector3{1, 1, 1};
566  return newVertex;
567  }

579  inline gcs::Vertex collapseEdge(gcs::Edge e) {
580  // First collect data
581  gcs::Vertex vertex1 = e.firstVertex(), vertex2 = e.secondVertex();
582  gc::Vector3 vertex1ForceMask = forces.forceMask[vertex1];
583  gc::Vector3 vertex2ForceMask = forces.forceMask[vertex2];
584  gc::Vector3 vertex1Pos = geometry.vpg->vertexPositions[vertex1];
585  gc::Vector3 vertex2Pos = geometry.vpg->vertexPositions[vertex2];
586  gc::Vector3 vertex1Vel = velocity[vertex1];
587  gc::Vector3 vertex2Vel = velocity[vertex2];
588  double vertex1GeoDist = geometry.geodesicDistance[vertex1];
589  double vertex2GeoDist = geometry.geodesicDistance[vertex2];
590  double vertex1Phi = proteinDensity[vertex1];
591  double vertex2Phi = proteinDensity[vertex2];
592  bool vertex1PointTracker = geometry.notableVertex[vertex1];
593  bool vertex2PointTracker = geometry.notableVertex[vertex2];
594 
595  // collapse the edge
596  gcs::Vertex newVertex = geometry.mesh->collapseEdgeTriangular(e);
597 
598  // vertex maybe void if edge was on a pinch triangle
599  if (newVertex.getIndex() != gc::INVALID_IND) {
600 
601  // update quantities
602  // Note: think about conservation of energy, momentum and angular
603  // momentum
604  geometry.vpg->vertexPositions[newVertex] =
605  ((gc::sum(vertex1ForceMask) < 2.5) || vertex1PointTracker)
606  ? vertex1Pos
607  : ((gc::sum(vertex2ForceMask) < 2.5) || vertex1PointTracker)
608  ? vertex2Pos
609  : (vertex1Pos + vertex2Pos) / 2;
610  // averageData(velocity, vertex1, vertex2, newVertex);
611  // averageData(geometry.geodesicDistance, vertex1, vertex2, newVertex);
612  // averageData(proteinDensity, vertex1, vertex2, newVertex);
613  velocity[newVertex] = 0.5 * (vertex1Vel + vertex2Vel);
614  geometry.geodesicDistance[newVertex] =
615  0.5 * (vertex1GeoDist + vertex2GeoDist);
616  proteinDensity[newVertex] = 0.5 * (vertex1Phi + vertex2Phi);
617  geometry.notableVertex[newVertex] =
618  vertex1PointTracker || vertex2PointTracker;
619  }
620  return newVertex;
621  }
