47 DLL_PUBLIC inline void sortVector(const Eigen::VectorXd &vec,
48  Eigen::VectorXd &sorted_vec,
49  Eigen::VectorXi &ind) {
50  ind = Eigen::VectorXi::LinSpaced(vec.size(), 0,
51  vec.size() - 1); //[0 1 2 3 ... N-1]
52  auto rule = [vec](int i, int j) -> bool { return vec(i) > vec(j); };
53  // regular expression, as a predicate of sort
54  std::sort(ind.data(), ind.data() + ind.size(), rule);
55  // The data member function returns a pointer to
56  // the first element of VectorXd, similar to
57  // begin()
58  sorted_vec.resize(vec.size());
59  for (int i = 0; i < vec.size(); i++) {
60  sorted_vec(i) = vec(ind(i));
61  }
62 }

137 DLL_PUBLIC inline bool hasOutlier(const Eigen::VectorXd &vec,
138  double threshold = 0.5) {
139  Eigen::VectorXd sorted_vec;
140  sortVector(vec, sorted_vec);
141  double r, l, range;
142  findRange(sorted_vec.data(), sorted_vec.size(), r, l);

198 DLL_PUBLIC inline double
199 signedVolumeFromFace(gcs::Face &f, gcs::VertexPositionGeometry &vpg) {
200  gc::Vector3 p[3];
201  size_t i = 0;
202  for (gcs::Vertex v : f.adjacentVertices()) {
203  p[i] = vpg.inputVertexPositions[v];
204  i++;
205  }
206  double v321 = p[2].x * p[1].y * p[0].z;
207  double v231 = p[1].x * p[2].y * p[0].z;
208  double v312 = p[2].x * p[0].y * p[1].z;
209  double v132 = p[0].x * p[2].y * p[1].z;
210  double v213 = p[1].x * p[0].y * p[2].z;
211  double v123 = p[0].x * p[1].y * p[2].z;
212 
213  return (-v321 + v231 + v312 - v132 - v213 + v123) / 6.0;
214 }

249 DLL_PUBLIC inline double getMeshVolume(gcs::ManifoldSurfaceMesh &mesh,
250  gcs::VertexPositionGeometry &vpg,
251  bool isFillHole = true) {
252  double volume = 0;
253  for (gcs::Face f : mesh.faces()) {
254  volume += signedVolumeFromFace(f, vpg);
255  }
256 
257  // fill hole for open mesh
258  if (mesh.hasBoundary()) {
259  if (isFillHole) {
260  for (gcs::BoundaryLoop bl : mesh.boundaryLoops()) {
261  gcs::Vertex theVertex = bl.halfedge().tailVertex();
262  for (gcs::Halfedge e : bl.adjacentHalfedges()) {
263  if (e.tailVertex() != theVertex && e.tipVertex() != theVertex) {
264  volume += signedVolumeFromFace(
265  std::vector<std::size_t>{e.tailVertex().getIndex(),
266  e.tipVertex().getIndex(),
267  theVertex.getIndex()},
268  vpg);
269  }
270  }
271  }
272  } else {
273  throw std::runtime_error("getMeshVolume: mesh is opened, not able to "
274  "compute enclosed volume unless filled holes!");
275  }
276  }
277  return volume;
278 }

305 DLL_PUBLIC inline gc::Vector3 cartesianToBarycentric(gc::Vector2 &v1,
306  gc::Vector2 &v2,
307  gc::Vector2 &v3,
308  gc::Vector2 &v) {
309  gc::Vector3 lambda;
310 
311  lambda.x = ((v2.y - v3.y) * (v.x - v3.x) + (v3.x - v2.x) * (v.y - v3.y)) /
312  ((v2.y - v3.y) * (v1.x - v3.x) + (v3.x - v2.x) * (v1.y - v3.y));
313  lambda.y = ((v3.y - v1.y) * (v.x - v3.x) + (v1.x - v3.x) * (v.y - v3.y)) /
314  ((v2.y - v3.y) * (v1.x - v3.x) + (v3.x - v2.x) * (v1.y - v3.y));
315  lambda.z = 1 - lambda.x - lambda.y;
316 
317  return lambda;
318 }

328 DLL_PUBLIC inline gc::Vector3
329 correspondBarycentricCoordinates(gc::Vector3 &baryCoords_,
330  gcs::Halfedge &firstHalfedge) {
331  size_t vertexInd = 0;
332  gc::Vector3 baryCoords;
333  for (gcs::Vertex v : firstHalfedge.face().adjacentVertices()) {
334  if (v == firstHalfedge.vertex()) {
335  baryCoords[vertexInd] = baryCoords_.x;
336  } else if (v == firstHalfedge.next().vertex()) {
337  baryCoords[vertexInd] = baryCoords_.y;
338  } else if (v == firstHalfedge.next().next().vertex()) {
339  baryCoords[vertexInd] = baryCoords_.z;
340  } else if (firstHalfedge.face().isBoundaryLoop()) {
341  throw std::runtime_error(
342  "correspondBarycentricCoordinates: face is on boundary!");
343  } else {
344  throw std::runtime_error(
345  "correspondBarycentricCoordinates: undefined behavior!");
346  }
347  vertexInd++;
348  }
349  return baryCoords;
350 }

386 DLL_PUBLIC inline Eigen::Matrix<double, Eigen::Dynamic, 3>
387 rowwiseCrossProduct(Eigen::Matrix<double, Eigen::Dynamic, 3> A,
388  Eigen::Matrix<double, Eigen::Dynamic, 3> B) {
389  Eigen::Matrix<double, Eigen::Dynamic, 3> C;
390  if (A.rows() != B.rows()) {
391  throw std::runtime_error("The input matrices must have same sizes!");
392  return C;
393  } else {
394  C.resize(A.rows(), 3);
395  for (size_t i = 0; i < A.rows(); i++) {
396  C.row(i) = A.row(i).cross(B.row(i));
397  }
398  return C;
399  }

628 DLL_PUBLIC inline gcs::Vertex
629 closestVertexToPt(gcs::SurfaceMesh &mesh, gcs::VertexPositionGeometry &vpg,
630  std::vector<double> position,
631  gcs::VertexData<double> &geodesicDistance,
632  double range = 1e10) {
633  gcs::Vertex theVertex;
634  double shorestDistance = 1e18;
635  bool isIntialized = !geodesicDistance.raw().isZero(0);
636  if (!isIntialized) {
637  std::cout << "\nWARNING: closestVertexToPt: geodesicDistance not "
638  "initialized, searching for all "
639  "vertices!"
640  << std::endl;
641  }
642  for (gcs::Vertex v : mesh.vertices()) {
643  if (geodesicDistance[v] > range) {
644  continue;
645  }
646  if (geodesicDistance[v] <= 0 && isIntialized) {
647  std::cout
648  << "\nWARNING: closestVertexToPt: geodesicDistance <= 0, may be "
649  "uninitialized/updated!"
650  << std::endl;
651  }
652  double distance;
653  if (position.size() == 2) {
654  distance = (gc::Vector2{vpg.inputVertexPositions[v].x,
655  vpg.inputVertexPositions[v].y} -
656  gc::Vector2{position[0], position[1]})
657  .norm();
658  } else if (position.size() == 3) {
659  distance = (vpg.inputVertexPositions[v] -
660  gc::Vector3{position[0], position[1], position[2]})
661  .norm();
662  } else {
663  throw std::runtime_error(
664  "closestVertexToPt: does not support non-2d/3d position vector!");
665  }
666  if (distance < shorestDistance) {
667  shorestDistance = distance;
668  theVertex = v;
669  }
670  }
671  return theVertex;
672 }

697 DLL_PUBLIC inline void
698 tanhDistribution(gcs::VertexPositionGeometry &vpg,
699  Eigen::Matrix<double, Eigen::Dynamic, 1> &distribution,
700  Eigen::Matrix<double, Eigen::Dynamic, 1> &distance,
701  double sharpness, std::vector<double> &axes) {
702  distribution.resize(distance.rows(), 1);
703  if (axes[0] == axes[1]) {
704  Eigen::MatrixXd radius_vec =
705  Eigen::MatrixXd::Constant(distance.rows(), 1, axes[0]);
706  distribution =
707  0.5 *
708  (1.0 + (sharpness * (radius_vec - distance)).array().tanh()).matrix();
709  } else {
710  double x, y, cos_t, radius;
711  for (size_t i = 0; i < distance.rows(); i++) {
712  x = vpg.inputVertexPositions[i].x;
713  y = vpg.inputVertexPositions[i].y;
714  cos_t = vpg.inputVertexPositions[i].x / sqrt(x * x + y * y);
715  radius = axes[0] * axes[1] /
716  sqrt((axes[0] * axes[0] - axes[1] * axes[1]) * cos_t * cos_t +
717  axes[1] * axes[1]);
718  distribution[i] = 0.5 * (1 + tanh(sharpness * (radius - distance[i])));
719  }
720  }
721 }

732 DLL_PUBLIC inline void
733 ellipticDistribution(gcs::VertexPositionGeometry &vpg,
734  Eigen::Matrix<double, Eigen::Dynamic, 1> &distribution,
735  Eigen::Matrix<double, Eigen::Dynamic, 1> &distance,
736  std::vector<double> &axes) {

739  distribution = (distance.array() < axes[0]).cast<double>();
740  } else {
741  double x, y, cos_t, radius;

746  radius = axes[0] * axes[1] /
747  sqrt((axes[0] * axes[0] - axes[1] * axes[1]) * cos_t * cos_t +
748  axes[1] * axes[1]);
749  distribution[i] = (double)(distance[i] < radius);
750  }
751  }
