53 DLL_PUBLIC inline void sortVector(const Eigen::VectorXd &vec,
54  Eigen::VectorXd &sorted_vec,
55  Eigen::VectorXi &ind) {
56  ind = Eigen::VectorXi::LinSpaced(vec.size(), 0,
57  vec.size() - 1); //[0 1 2 3 ... N-1]
58  auto rule = [vec](int i, int j) -> bool { return vec(i) > vec(j); };
59  // regular expression, as a predicate of sort
60  std::sort(ind.data(), ind.data() + ind.size(), rule);
61  // The data member function returns a pointer to
62  // the first element of VectorXd, similar to
63  // begin()
64  sorted_vec.resize(vec.size());
65  for (int i = 0; i < vec.size(); i++) {
66  sorted_vec(i) = vec(ind(i));
67  }
68 }

143 DLL_PUBLIC inline bool hasOutlier(const Eigen::VectorXd &vec,
144  double threshold = 0.5) {
145  Eigen::VectorXd sorted_vec;
146  sortVector(vec, sorted_vec);
147  double r, l, range;
148  findRange(sorted_vec.data(), sorted_vec.size(), r, l);
149  range = r - l;
150  return (sorted_vec[0] - r > threshold * range) ||
151  (l - sorted_vec[sorted_vec.size() - 1] > threshold * range);
152 }

163 DLL_PUBLIC inline EigenVectorX1_T<bool> outlierMask(const EigenVectorX1d &vec,
164  double threshold = 0.5,
165  bool negate = false) {
166  EigenVectorX1_T<bool> mask(vec.rows(), 1);
167  EigenVectorX1d sorted_vec(vec.rows(), 1);
168  sortVector(vec, sorted_vec);
169  double r, l, range;
170  findRange(sorted_vec.data(), sorted_vec.size(), r, l);
171  range = r - l;
172  if (negate) {
173  // outlier: true
174  for (Eigen::Index i = 0; i < vec.rows(); i++) {
175  mask[i] = ((vec[i] - r < threshold * range) &&
176  (l - vec[i] < threshold * range));
177  }
178  } else {
179  // outlier: false
180  for (Eigen::Index i = 0; i < vec.rows(); i++) {
181  mask[i] = ((vec[i] - r > threshold * range) ||
182  (l - vec[i] > threshold * range));
183  }
184  }
185  return mask;
186 }

235 DLL_PUBLIC inline double signedVolumeFromFace(gc::Vector3 &p0, gc::Vector3 &p1,
236  gc::Vector3 &p2) {
237  double v321 = p2.x * p1.y * p0.z;
238  double v231 = p1.x * p2.y * p0.z;
239  double v312 = p2.x * p0.y * p1.z;
240  double v132 = p0.x * p2.y * p1.z;
241  double v213 = p1.x * p0.y * p2.z;
242  double v123 = p0.x * p1.y * p2.z;
243  return (-v321 + v231 + v312 - v132 - v213 + v123) / 6.0;
244 }

274 DLL_PUBLIC inline double getMeshVolume(gcs::ManifoldSurfaceMesh &mesh,
275  gcs::VertexPositionGeometry &vpg,
276  bool isFillHole = true) {
277  double volume = 0;
278 
279  // Throw error or fill hole for open mesh
280  if (mesh.hasBoundary()) {
281  if (isFillHole) {
282  for (gcs::BoundaryLoop bl : mesh.boundaryLoops()) {
283  gcs::Vertex theVertex = bl.halfedge().tailVertex();
284  for (gcs::Halfedge e : bl.adjacentHalfedges()) {
285  if (e.tailVertex() != theVertex && e.tipVertex() != theVertex) {
286  volume += signedVolumeFromFace(e.tailVertex(), e.tipVertex(),
287  theVertex, vpg);
288  }
289  }
290  }
291  } else {
292  mem3dg_runtime_error("Mesh is opened, not able to ",
293  "compute enclosed volume unless filled holes!");
294  }
295  }
296 
297  for (gcs::Face f : mesh.faces()) {
298  volume += signedVolumeFromFace(f, vpg);
299  }
300 
301  return volume;
302 }

329 DLL_PUBLIC inline gc::Vector3 cartesianToBarycentric(gc::Vector2 &v1,
330  gc::Vector2 &v2,
331  gc::Vector2 &v3,
332  gc::Vector2 &v) {
333  gc::Vector3 lambda{
334  ((v2.y - v3.y) * (v.x - v3.x) + (v3.x - v2.x) * (v.y - v3.y)) /
335  ((v2.y - v3.y) * (v1.x - v3.x) + (v3.x - v2.x) * (v1.y - v3.y)), // X
336  ((v3.y - v1.y) * (v.x - v3.x) + (v1.x - v3.x) * (v.y - v3.y)) /
337  ((v2.y - v3.y) * (v1.x - v3.x) + (v3.x - v2.x) * (v1.y - v3.y)), // Y
338  0};
339  lambda.z = 1 - lambda.x - lambda.y;
340 
341  return lambda;
342 }

352 DLL_PUBLIC inline gc::Vector3
353 correspondBarycentricCoordinates(const gc::Vector3 baryCoords_,
354  const gcs::Halfedge &firstHalfedge) {
355  std::size_t vertexInd = 0;
356  gc::Vector3 baryCoords;
357  for (gcs::Vertex v : firstHalfedge.face().adjacentVertices()) {
358  if (v == firstHalfedge.vertex()) {
359  baryCoords[vertexInd] = baryCoords_.x;
360  } else if (v == firstHalfedge.next().vertex()) {
361  baryCoords[vertexInd] = baryCoords_.y;
362  } else if (v == firstHalfedge.next().next().vertex()) {
363  baryCoords[vertexInd] = baryCoords_.z;
364  } else if (firstHalfedge.face().isBoundaryLoop()) {
365  mem3dg_runtime_error("Face is on boundary!");
366  } else {
367  mem3dg_runtime_error("Undefined behavior");
368  }
369  vertexInd++;
370  }
371  return baryCoords;
372 }

542 inline double gaussianDistribution(const double &distance,
543  const double &stdDev) {
544  return exp(-distance * distance / (2 * stdDev * stdDev));
545 }
546 inline double
547 gaussianDistribution(const double &distance,
548  const gc::Vector3 &vertexPositionFromPtInd,
549  const std::array<gc::Vector3, 2> &tangentBasis,
550  const std::array<double, 2> &stdDev) {
551  double x = gc::dot(vertexPositionFromPtInd, tangentBasis[0]);
552  double y = gc::dot(vertexPositionFromPtInd, tangentBasis[1]);
553  double cos_t = 1.0;
554  if (!(x == 0 && y == 0)) {
555  cos_t = x / sqrt(x * x + y * y);
556  }
557  return exp(
558  -((distance * distance * cos_t * cos_t / (2 * stdDev[0] * stdDev[0])) +
559  (distance * distance * (1 - cos_t * cos_t) /
560  (2 * stdDev[1] * stdDev[1]))));
561 }

563 gaussianDistribution(Eigen::Ref<EigenVectorX1d> distribution,

593 inline double tanhDistribution(const double &distance, const double &sharpness,
594  const double &ax) {
595  return 0.5 * (1.0 + tanh(sharpness * (ax - distance)));
596 }
597 inline double tanhDistribution(const double &distance,
598  const gc::Vector3 &vertexPositionFromPtInd,
599  const std::array<gc::Vector3, 2> &tangentBasis,
600  const double &sharpness,
601  const std::array<double, 2> &axes) {
602  double x = gc::dot(vertexPositionFromPtInd, tangentBasis[0]);
603  double y = gc::dot(vertexPositionFromPtInd, tangentBasis[1]);
604  double cos_t = 1.0;
605  if (!(x == 0 && y == 0)) {
606  cos_t = x / sqrt(x * x + y * y);
607  }
608  double ax = axes[0] * axes[1] /
609  sqrt((axes[0] * axes[0] - axes[1] * axes[1]) * cos_t * cos_t +
610  axes[1] * axes[1]);
611  return 0.5 * (1.0 + tanh(sharpness * (ax - distance)));
612 }
613 DLL_PUBLIC inline void
614 tanhDistribution(Eigen::Ref<EigenVectorX1d> distribution,
615  const Eigen::Ref<const EigenVectorX1d> &distances,
616  const double sharpness, const double &ax) {
617  Eigen::MatrixXd radius_vec =
618  Eigen::MatrixXd::Constant(distances.rows(), 1, ax);
619  distribution =
620  0.5 *
621  (1.0 + (sharpness * (radius_vec - distances)).array().tanh()).matrix();
622 }

696 DLL_PUBLIC inline double
697 jumpDistribution(const double &distance,
698  const gc::Vector3 &vertexPositionFromPtInd,
699  const std::array<gc::Vector3, 2> &tangentBasis,
700  const std::array<double, 2> &axes) {
701  double x = gc::dot(vertexPositionFromPtInd, tangentBasis[0]);
702  double y = gc::dot(vertexPositionFromPtInd, tangentBasis[1]);
703  double cos_t = 1.0;
704  if (!(x == 0 && y == 0)) {
705  cos_t = x / sqrt(x * x + y * y);
706  }
707  double ax = axes[0] * axes[1] /
708  sqrt((axes[0] * axes[0] - axes[1] * axes[1]) * cos_t * cos_t +
709  axes[1] * axes[1]);
710  return (double)(distance < ax);
711 }
