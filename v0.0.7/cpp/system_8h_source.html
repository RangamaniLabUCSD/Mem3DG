547  inline gcs::Vertex splitEdge(gcs::Edge e) {
548  gcs::Vertex vertex1 = e.firstVertex(), vertex2 = e.secondVertex();
549  gc::Vector3 vertex1Pos = geometry.vpg->vertexPositions[vertex1];
550  gc::Vector3 vertex2Pos = geometry.vpg->vertexPositions[vertex2];
551  gc::Vector3 vertex1Vel = velocity[vertex1];
552  gc::Vector3 vertex2Vel = velocity[vertex2];
553  double vertex1GeoDist = geometry.geodesicDistance[vertex1];
554  double vertex2GeoDist = geometry.geodesicDistance[vertex2];
555  double vertex1Phi = proteinDensity[vertex1];
556  double vertex2Phi = proteinDensity[vertex2];
557 
558  // bool vertex1PointTracker = geometry.notableVertex[vertex1];
559  // bool vertex2PointTracker = geometry.notableVertex[vertex2];
560 
561  // split the edge
562  gcs::Vertex newVertex = geometry.mesh->splitEdgeTriangular(e).vertex();
563 
564  // update quantities
565  // Note: think about conservation of energy, momentum and angular
566  // momentum
567  // averageData(geometry.vpg->inputVertexPositions, vertex1, vertex2,
568  // newVertex); averageData(velocity, vertex1, vertex2, newVertex);
569  // averageData(geometry.geodesicDistance, vertex1, vertex2, newVertex);
570  // averageData(proteinDensity, vertex1, vertex2, newVertex);
571  geometry.vpg->vertexPositions[newVertex] = 0.5 * (vertex1Pos + vertex2Pos);
572  velocity[newVertex] = 0.5 * (vertex1Vel + vertex2Vel);
573  geometry.geodesicDistance[newVertex] =
574  0.5 * (vertex1GeoDist + vertex2GeoDist);
575  proteinDensity[newVertex] = 0.5 * (vertex1Phi + vertex2Phi);
576  geometry.notableVertex[newVertex] = false;
577  forces.forceMask[newVertex] = gc::Vector3{1, 1, 1};
578  return newVertex;
579  }

591  inline gcs::Vertex collapseEdge(gcs::Edge e) {
592  // First collect data
593  gcs::Vertex vertex1 = e.firstVertex(), vertex2 = e.secondVertex();
594  gc::Vector3 vertex1ForceMask = forces.forceMask[vertex1];
595  gc::Vector3 vertex2ForceMask = forces.forceMask[vertex2];
596  gc::Vector3 vertex1Pos = geometry.vpg->vertexPositions[vertex1];
597  gc::Vector3 vertex2Pos = geometry.vpg->vertexPositions[vertex2];
598  gc::Vector3 vertex1Vel = velocity[vertex1];
599  gc::Vector3 vertex2Vel = velocity[vertex2];
600  double vertex1GeoDist = geometry.geodesicDistance[vertex1];
601  double vertex2GeoDist = geometry.geodesicDistance[vertex2];
602  double vertex1Phi = proteinDensity[vertex1];
603  double vertex2Phi = proteinDensity[vertex2];
604  bool vertex1PointTracker = geometry.notableVertex[vertex1];
605  bool vertex2PointTracker = geometry.notableVertex[vertex2];
606 
607  // collapse the edge
608  gcs::Vertex newVertex = geometry.mesh->collapseEdgeTriangular(e);
609 
610  // vertex maybe void if edge was on a pinch triangle
611  if (newVertex.getIndex() != gc::INVALID_IND) {
612 
613  // update quantities
614  // Note: think about conservation of energy, momentum and angular
615  // momentum
616  geometry.vpg->vertexPositions[newVertex] =
617  ((gc::sum(vertex1ForceMask) < 2.5) || vertex1PointTracker)
618  ? vertex1Pos
619  : ((gc::sum(vertex2ForceMask) < 2.5) || vertex2PointTracker)
620  ? vertex2Pos
621  : (vertex1Pos + vertex2Pos) / 2;
622  // averageData(velocity, vertex1, vertex2, newVertex);
623  // averageData(geometry.geodesicDistance, vertex1, vertex2, newVertex);
624  // averageData(proteinDensity, vertex1, vertex2, newVertex);
625  velocity[newVertex] = 0.5 * (vertex1Vel + vertex2Vel);
626  geometry.geodesicDistance[newVertex] =
627  0.5 * (vertex1GeoDist + vertex2GeoDist);
628  proteinDensity[newVertex] = 0.5 * (vertex1Phi + vertex2Phi);
629  geometry.notableVertex[newVertex] =
630  vertex1PointTracker || vertex2PointTracker;
631  }
632  return newVertex;
633  }
