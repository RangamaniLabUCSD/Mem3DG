174  Force(gcs::SurfaceMesh &mesh_, gcs::VertexPositionGeometry &vpg_,
175  gcs::VertexPositionGeometry &refVpg_,
176  gcs::RichSurfaceMeshData &richData_, Parameters &p, bool isProtein_ = false,
177  bool isTuftedLaplacian_ = false, double mollifyFactor_ = 1e-6,
178  bool isVertexShift_ = false)
179  : mesh(mesh_), vpg(vpg_), richData(richData_), refVpg(refVpg_),
180  P(p), isTuftedLaplacian(isTuftedLaplacian_), isProtein(isProtein_), mollifyFactor(mollifyFactor_),
181  isVertexShift(isVertexShift_), bendingPressure(mesh_, {0, 0, 0}),
182  insidePressure(mesh_, {0, 0, 0}), capillaryPressure(mesh_, {0, 0, 0}),
183  externalPressure(mesh_, {0, 0, 0}), regularizationForce(mesh_, {0,0,0}),
184  stochasticForce(mesh_, {0, 0, 0}), dampingForce(mesh_, {0, 0, 0}),
185  proteinDensity(mesh_, 0), vel(mesh_, {0, 0, 0}) {
186 
187  // Initialize RNG
188  pcg_extras::seed_seq_from<std::random_device> seed_source;
189  rng = pcg32(seed_source);
190 
191  // GC computed properties
192  vpg.requireFaceNormals();
193  vpg.requireVertexGalerkinMassMatrix();
194  vpg.requireVertexLumpedMassMatrix();
195  vpg.requireCotanLaplacian();
196  vpg.requireFaceAreas();
197  vpg.requireVertexIndices();
198  vpg.requireVertexGaussianCurvatures();
199  vpg.requireFaceIndices();
200  vpg.requireEdgeLengths();
201  vpg.requireVertexNormals();
202  vpg.requireVertexDualAreas();
203  vpg.requireCornerAngles();
204 
205  refVpg.requireFaceAreas();
206  refVpg.requireEdgeLengths();
207 
208  // Initialize the geodesic distance from ptInd
209  geodesicDistanceFromAppliedForce =
210  heatMethodDistance(vpg, mesh.vertex(P.ptInd));
211  auto &dist_e = geodesicDistanceFromAppliedForce.raw();
212 
213  // Initialize the external pressure magnitude distribution
214  gaussianDistribution(externalPressureMagnitude, dist_e,
215  dist_e.maxCoeff() / P.conc);
216  externalPressureMagnitude *= P.Kf;
217 
218  // Initialize the spontaneous curvature distribution
219  if (isProtein) {
220  proteinDensity.raw().setZero();
221  H0.setZero(mesh.nVertices(), 1);
222  } else {
223  tanhDistribution(H0, dist_e, P.sharpness, P.r_H0);
224  H0 *= P.H0;
225  }
226 
227  // Initialize the mask on choosing integration vertices based on geodesic
228  // distance from the local external force location on the reference geometry
229  mask = (heatMethodDistance(refVpg, mesh.vertex(P.ptInd)).raw().array() <
230  P.radius)
231  .matrix();
232 
233  // Regularize the vetex position geometry if needed
234  if (isVertexShift) {
235  vertexShift(mesh, vpg, mask);
236  update_Vertex_positions();
237  }
238 
239  // Initialize the mass and conformal Laplacian matrix
240  if (isTuftedLaplacian) {
241  getTuftedLaplacianAndMass(M, L, mesh, vpg, mollifyFactor);
242  } else {
243  M = vpg.vertexLumpedMassMatrix;
244  L = vpg.cotanLaplacian;
245  }
246 
247  // Initialize the inverse mass matrix
248  M_inv = (1 / (M.diagonal().array())).matrix().asDiagonal();
249  // Alternatively, use the Galerkin mass matrix
250  // M = vpg.vertexGalerkinMassMatrix;
251  // Eigen::SparseLU<Eigen::SparseMatrix<double>> solver;
252  // solver.compute(vpg.vertexGalerkinMassMatrix);
253  // std::size_t n = mesh.nVertices();
254  // Eigen::SparseMatrix<double> I(n, n);
255  // I.setIdentity();
256  // M_inv = solver.solve(I);
257 
258  // Initialize target face/surface areas
259  targetFaceAreas = refVpg.faceAreas.reinterpretTo(mesh);
260  targetSurfaceArea = targetFaceAreas.raw().sum();
261 
262  // Initialize edge length
263  targetEdgeLengths = refVpg.edgeLengths.reinterpretTo(mesh);
264 
265  // Initialize reference volume
266  if (mesh.hasBoundary()) {
267  refVolume = 0.0;
268  } else {
269  refVolume = std::pow(targetSurfaceArea / M_PI / 4, 1.5) * (4 * M_PI / 3);
270  }
271 
272  // Initialize surface area
273  surfaceArea = vpg.faceAreas.raw().sum();
274 
275  // Initialize volume
276  for (gcs::Face f : mesh.faces()) {
277  volume += signedVolumeFromFace(
278  f, vpg, vpg.inputVertexPositions[mesh.vertex(P.ptInd)]);
279  }
280 
281  // Initialize the vertex position of the last iteration
282  pastPositions = vpg.inputVertexPositions;
283  }

303  vpg.unrequireVertexDualAreas();
304  vpg.unrequireCornerAngles();
305  }
