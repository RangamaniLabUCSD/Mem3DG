31 inline void show_backtrace() {
32  unw_cursor_t cursor;
33  unw_context_t context;
34 
35  // Initialize cursor to current frame for local unwinding.
36  unw_getcontext(&context);
37  unw_init_local(&cursor, &context);
38 
39  // Unwind frames one by one, going up the frame stack.
40  while (unw_step(&cursor) > 0) {
41  unw_word_t offset, ip;
42  // get instruction pointer
43  unw_get_reg(&cursor, UNW_REG_IP, &ip);
44  if (ip == 0) {
45  break;
46  }
47  printf("0x%lx:", ip);
48 
49  char sym[256];
50  if (unw_get_proc_name(&cursor, sym, sizeof(sym), &offset) == 0) {
51  printf(" (%s+0x%lx)\n", sym, offset);
52  } else {
53  printf(" -- error: unable to obtain symbol name for this frame\n");
54  }
55  }
56 }
57 #else
58 #define show_backtrace() ((void)0)
59 #endif
