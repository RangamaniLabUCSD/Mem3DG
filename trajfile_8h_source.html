300  TrajFile(const std::string &filename, gcs::SurfaceMesh &mesh,
301  gcs::VertexPositionGeometry &refVpg, const NcFile::FileMode fMode)
302  : filename(filename), // fd(new NcFile(filename, fMode)),
303  writeable(true) {
304 
305  fd = new NcFile(filename, fMode);
306  // initialize data
307  fd->putAtt(CONVENTIONS_NAME, CONVENTIONS_VALUE);
308  fd->putAtt(CONVENTIONS_VERSION_NAME, CONVENTIONS_VERSION_VALUE);
309 
310  frame_dim = fd->addDim(FRAME_NAME);
311  npolygons_dim = fd->addDim(NPOLYGONS_NAME, mesh.nFaces());
312  nvertices_dim = fd->addDim(NVERTICES_NAME, mesh.nVertices());
313  spatial_dim = fd->addDim(SPATIAL_DIMS_NAME, SPATIAL_DIMS);
314  polygon_order_dim = fd->addDim(POLYGON_ORDER_NAME, POLYGON_ORDER);
315 
316  // Initialize topology data block
317  topology =
318  fd->addVar(TOPO_VAR, nc::ncUint, {npolygons_dim, polygon_order_dim});
319 
320  // Populate topology data
321  Eigen::Matrix<std::uint32_t, Eigen::Dynamic, 3, Eigen::RowMajor>
322  faceMatrix = getFaceVertexMatrix(mesh);
323  std::uint32_t *topodata = faceMatrix.data();
324  topology.putVar(topodata);
325 
326  // Initialize reference coordinate data block
327  refcoord = fd->addVar(REFCOORD_VAR, netCDF::ncDouble,
328  {nvertices_dim, spatial_dim});
329 
330  // Populate reference coordinate data
331  double *refcoorddata;
332  refcoorddata = gc::EigenMap<double, 3>(refVpg.inputVertexPositions).data();
333  refcoord.putVar(refcoorddata);
334 
335  time_var = fd->addVar(TIME_VAR, netCDF::ncDouble, {frame_dim});
336  time_var.putAtt(UNITS, TIME_UNITS);
337 
338  coord_var = fd->addVar(COORD_VAR, netCDF::ncDouble,
339  {frame_dim, nvertices_dim, spatial_dim});
340  coord_var.putAtt(UNITS, LEN_UNITS);
341 
342  vel_var = fd->addVar(VEL_VAR, netCDF::ncDouble,
343  {frame_dim, nvertices_dim, spatial_dim});
344 
345  meancurve_var =
346  fd->addVar(MEANCURVE_VAR, netCDF::ncDouble, {frame_dim, nvertices_dim});
347 
348  sponcurve_var =
349  fd->addVar(SPONCURVE_VAR, netCDF::ncDouble, {frame_dim, nvertices_dim});
350 
351  externpress_var = fd->addVar(EXTERNPRESS_VAR, netCDF::ncDouble,
352  {frame_dim, nvertices_dim});
353 
354  physpress_var =
355  fd->addVar(PHYSPRESS_VAR, netCDF::ncDouble, {frame_dim, nvertices_dim});
